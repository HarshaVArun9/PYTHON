# Password Generator

import random
import string

length = int(input("Enter password length: "))

characters = string.ascii_letters + string.digits + string.punctuation
password = ""

for i in range(length):
    password += random.choice(characters)

print("Generated Password:", password)


"""
Notes:

- string module provides letters, digits, and special characters.
- random.choice() selects a random character from the given set.
- A loop is used to generate characters based on required length.
- Password is built by appending random characters.
- Demonstrates loops, modules, and randomness.
"""
